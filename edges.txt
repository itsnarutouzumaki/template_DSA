void edges(map<int,vector<int>> &m,vector<int> &e)
{
    for(int i=0;i<e.size();i++)
    {
      m[e[i][0]].push_back(e[i][1]);
      m[e[i][1]].push_back(e[i][0]);
    }
}

// for weighted

void edges(map<int,vector<pair<int,int>>> &m,vector<int> &e)
{
    for(int i=0;i<e.size();i++)
    {
      m[e[i][0]].push_back({e[i][1],e[i][2]});
      m[e[i][1]].push_back({e[i][0],e[i][2]});
    }
}
